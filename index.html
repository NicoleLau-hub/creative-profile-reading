    function scoreAnswers(answers) {

      var counts = { fight: 0, flight: 0, freeze: 0, fawn: 0, collapse: 0 };

      ["q1","q2","q3","q4","q5","q7"].forEach(function(qid) {

        var found = STEPS.find(function(item) { return item.id === qid; });

        if (!found) return;

        var chosen = found.options.find(function(opt) { return opt.id === answers[qid]; });

        if (chosen && chosen.pattern) counts[chosen.pattern]++;

      });

      var sorted = Object.entries(counts).sort(function(a, b) { return b[1] - a[1]; });

      var topCount = sorted[0][1];

      var topPatterns = sorted.filter(function(e) { return e[1] === topCount; }).map(function(e) { return e[0]; });

      var pattern;

      if (topPatterns.length === 1) {

        pattern = topPatterns[0];

      } else {

        var q7found = STEPS.find(function(item) { return item.id === "q7"; });

        var q7chosen = q7found ? q7found.options.find(function(opt) { return opt.id === answers.q7; }) : null;

        pattern = (q7chosen && q7chosen.pattern && topPatterns.includes(q7chosen.pattern)) ? q7chosen.pattern : topPatterns[0];

      }

      var segment = "Mentorship for Consistency";

      return { pattern: pattern, counts: counts, segment: segment, offerProgram: true };

    }

      function parseSection(text, heading) {

        if (!text) return "";

        var lines = text.split("\n");

        var inside = false;

        var out = [];

        for (var i = 0; i < lines.length; i++) {

          var line = lines[i];

          if (line.trim().toUpperCase() === heading.toUpperCase()) { inside = true; continue; }

          if (inside) {

            if (line.trim().length > 4 && line.trim() === line.trim().toUpperCase() && /^[A-Z]/.test(line.trim())) break;

            out.push(line);

          }

        }

        return out.join("\n").trim();

      }

    function App() {

      var stepIndexState = useState(0);

      var stepIndex = stepIndexState[0], setStepIndex = stepIndexState[1];

      var answersState = useState({});

      var answers = answersState[0], setAnswers = answersState[1];


      var step = STEPS[stepIndex];

      var qStep = (step && (step.type === "single" || step.type === "icon_cards")) ? step : null;


      function setAnswer(id, val) {

        setAnswers(function(prev) {

          var next = Object.assign({}, prev);

          next[id] = val;

          return next;

        });

      }

      function next()  { setStepIndex(function(i) { return i + 1; }); window.scrollTo(0, 0); }

      function back()  { setStepIndex(function(i) { return i - 1; }); window.scrollTo(0, 0); }

      function reset() { setStepIndex(0); setAnswers({}); window.scrollTo(0, 0); }


      return h("div", { className: "quiz-root" },

        h(TopBar, { currentStepId: step.id, currentStep: qStep ? qStep.step : null, totalSteps: qStep ? qStep.total : null }),

        step.type === "welcome"    ? h(WelcomeStep,   { onNext: next }) : null,

        step.type === "single"     ? h(QuestionStep,  { step: step, answers: answers, setAnswer: setAnswer, onNext: next, onBack: back }) : null,

        step.type === "icon_cards" ? h(IconCardStep,  { step: step, answers: answers, setAnswer: setAnswer, onNext: next, onBack: back }) : null,

        step.type === "confirm"    ? h(ConfirmStep,   { answers: answers, onBack: back, onSubmit: next }) : null,

        step.type === "result"     ? h(ResultStep,    { answers: answers, onReset: reset }) : null

    }

  })();
